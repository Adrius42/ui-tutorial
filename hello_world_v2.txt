3.) Change to the UI editor (third symbol / looking like a square)
4.) Push '(No Name)' and change the heading of the View to 'HW2'
5.) Change the text of label1 to 'Hello' (remember: Push the Label until the menu apears and choose "Attributes...")
6.) Now change to the code view and add the missing lines
# coding: utf-8

import ui, time

def change_label(label):
  if label.text == 'Hello':
    label.text = 'World'
  else:
    label.text = 'Hello'

view = ui.load_view('hello_world_v2')
my_label = view['label1']
view.present('fullscreen')

while True:
  time.sleep(1)
  change_label(my_label)
